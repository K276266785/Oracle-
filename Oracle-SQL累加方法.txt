SELECT A.UserName,
       A.Month,
       MAX(A.Salary) as Salary,
       SUM (B.Salary) as Accumulation
       FROM User_Salary A inner join User_Salary B 
       ON A.UserName = B.UserName
       where B.Month <= A.Month
       group by A.UserName,A.Month 
       order by A.UserName,A.Month    
go
